    def abspath(path):
    ""Returns the absolute version of a path.""

    if path: #Empty path must return current working directory.
        try:
            path = _getfullpathname(path)
        except WindowsError:
            pass # Bad path - return unchanged
    elif isinstance(path, unicode):
        path = os.getcwdu()
    else:
        path = os.getcwd()
    return normpath(path)
